function generarPresupuesto() { const litros = parseFloat(document.getElementById('litros').value); const materialOpcion = document.getElementById('material').value; const conCastillos = document.getElementById('castillos').checked; if (!litros || litros < 1000) return alert("Ingrese capacidad válida."); // --- PARÁMETROS TÉCNICOS --- const lado = Math.pow(litros/1000, 1/3); const areaLosa = (lado * lado) * 1.2; const mlNervio = lado; // Un nervio central a lo largo del claro // Costos Unitarios (Puebla 2026) const precioMuros = { "ladrillo": 4800, "block": 5500, "concreto": 8500 }; const costoLosaM2 = 1250; // Vigueta, bovedilla y malla const costoCastillo = 1050; const costoNervioML = 210; // 2 varillas 3/8" + grapas alambrón #2 // --- CÁLCULO --- const subtotalMuros = (litros / 1000) * precioMuros[materialOpcion]; const subtotalLosa = areaLosa * costoLosaM2; const subtotalNervio = mlNervio * costoNervioML; const subtotalCastillos = conCastillos ? 4 * costoCastillo : 0; // Insumos auxiliares (Clavo, madera, alambre, fletes) const gastosFijos = 8500; const total = subtotalMuros + subtotalLosa + subtotalNervio + subtotalCastillos + gastosFijos; const formato = (n) => new Intl.NumberFormat('es-MX', { style: 'currency', currency: 'MXN' }).format(n); document.getElementById('monto').innerText = formato(total); // Detalle Técnico para el Arq. Desglose de Obra:
Materiales y Acero: ${formato(total * 0.58)}
